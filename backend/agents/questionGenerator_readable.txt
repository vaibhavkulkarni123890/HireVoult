const fetch = require('node-fetch');

function removeDuplicateMCQs(questions) {
  const seen = new Set();
  return questions.filter(q => {
    if (q.type !== 'mcq') return true;
    
    // Extract key concept words from question
    const key = q.question
      .toLowerCase()
      .replace(/what is the (primary|main) (purpose|benefit|goal|difference|function) of/g, '')
      .replace(/in (distributed systems|cloud computing|databases|algorithm design)/g, '')
      .replace(/why (is|do|should) we use/g, '')
      .trim()
      .split(' ')
      .slice(0, 5)
      .join(' ');
    
    if (seen.has(key)) return false; // Remove duplicate
    seen.add(key);
    return true;
  });
}

function sanitizeTestCases(questions) {
  if (!Array.isArray(questions)) return [];
  const mcqs = removeDuplicateMCQs(questions); // Apply semantic deduplication
  
  return mcqs.map(q => {
    // Remove "Why this question:" prefix if AI added it, to prevent double-prefix issues
    if (q.reasoning) {
        q.reasoning = q.reasoning.replace(/^(Why this question:|reasoning:)\s*/i, '').trim();
    }

    if (q.type !== 'coding') return q;
    
    // Validate and fix test cases
    if (q.testCases) {
      q.testCases = q.testCases.map(tc => {
        // Ensure input is proper object not string
        if (typeof tc.input === 'string') {
          try {
            const parsed = JSON.parse(tc.input);
            if (typeof parsed === 'object') tc.input = parsed;
          } catch(e) {}
        }
        return tc;
      });
    }

    // Ensure reasoning is present and follows format
    if (!q.reasoning || q.reasoning.length < 20) {
      q.reasoning = `Selected for this role based on technical requirements in ${q.title || 'engineering'}.`;
    }

    return q;
  });
}

/**
 * Generate follow-up theory questions based on candidate's submitted code
 */
async function generateTheoryFromCode(code, language, problemStatement) {
  const groqKey = process.env.GROQ_API_KEY;
  if (!groqKey) return [];
  const prompt = `Code submitted:\n\n${code}\n\nGenerate 2 technical follow-up questions about this code. JSON: { "questions": [{ "type": "theory", "question": "...", "rubric": "...", "points": 10, "timeLimit": 180 }] }`;
  try {
    const response = await fetch("https://api.groq.com/openai/v1/chat/completions", {
      method: "POST",
      headers: { "Content-Type": "application/json", "Authorization": `Bearer ${groqKey}` },
      body: JSON.stringify({ model: "llama-3.3-70b-versatile", messages: [{ role: "user", content: prompt }], response_format: { type: "json_object" } })
    });
    const data = await response.json();
    return JSON.parse(data.choices[0].message.content).questions || [];
  } catch (e) { return []; }
}
